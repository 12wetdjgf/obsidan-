void insert(int[] nums,int num,int index){
	for (int i = nums.length - 1; i > index; i --){
		nums[i] = nums[i - 1];
		}
		nums[index] = num;
}

void traverse(int[] nums){
	int count = 0;
	for (int i = 0; i < nums.length; i++){
		count += nums[i];
	}
	for (int num : nums){
		count += num;
	}
}

int find(int[] nums, int target){
	for (int i = 0; i < nums.length; i ++){
		if (nums[i] == target)
			return i;
	}
	return -1
}
