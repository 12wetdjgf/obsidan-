## 🎯 模板3：TodoWrite系统（s03）

添加了 `TodoManager` 和自动提醒机制。

```python
class TodoManager:
    def __init__(self):
        self.items = []
        self.rounds_since_update = 0
    
    def update(self, items: list) -> str:
        """更新todo列表"""
        validated = []
        in_progress_count = 0
        
        for item in items:
            status = item.get("status", "pending")
            if status == "in_progress":
                in_progress_count += 1
            validated.append({
                "id": item["id"],
                "text": item["text"],
                "status": status
            })
        
        if in_progress_count > 1:
            raise ValueError("❌ 只能有一个任务是 in_progress")
        
        self.items = validated
        self.rounds_since_update = 0
        return self.render()
    
    def render(self) -> str:
        """渲染todo列表"""
        lines = ["📋 当前任务列表："]
        for item in self.items:
            status_icon = {
                "pending": "[ ]",
                "in_progress": "[>]",
                "completed": "[x]"
            }.get(item["status"], "[ ]")
            lines.append(f"{status_icon} {item['id']}: {item['text']}")
        return "\n".join(lines)
    
    def increment_rounds(self):
        """增加轮数计数"""
        self.rounds_since_update += 1
    
    def should_remind(self) -> bool:
        """是否应该提醒"""
        return self.rounds_since_update >= 3

# 在 TOOLS 中添加
TODO_TOOL = {
    "name": "todo",
    "description": "更新任务列表。每个任务有id、text和status。",
    "input_schema": {
        "type": "object",
        "properties": {
            "items": {
                "type": "array",
                "items": {
                    "type": "object",
                    "properties": {
                        "id": {"type": "string"},
                        "text": {"type": "string"},
                        "status": {
                            "type": "string",
                            "enum": ["pending", "in_progress", "completed"]
                        }
                    },
                    "required": ["id", "text", "status"]
                }
            }
        },
        "required": ["items"]
    }
}

# 在 agent_loop 中修改
def agent_loop(query: str):
    messages = [{"role": "user", "content": query}]
    todo_manager = TodoManager()
    print(f"\n🤖 用户: {query}\n")
    
    while True:
        # 检查是否需要提醒
        if todo_manager.should_remind() and messages:
            last = messages[-1]
            if last["role"] == "user" and isinstance(last.get("content"), list):
                last["content"].insert(0, {
                    "type": "text",
                    "text": "⚠️ <reminder>更新你的todo列表。</reminder>"
                })
        
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            system=SYSTEM,
            messages=messages,
            tools=TOOLS,  # 包含 TODO_TOOL
            max_tokens=8000,
        )
        
        messages.append({"role": "assistant", "content": response.content})
        
        if response.stop_reason != "tool_use":
            print(f"✅ 完成！\n{todo_manager.render()}")
            return
        
        results = []
        for block in response.content:
            if block.type == "tool_use":
                if block.name == "todo":
                    output = todo_manager.update(block.input["items"])
                    print(f"📋 {output}\n")
                else:
                    handler = TOOL_HANDLERS.get(block.name)
                    output = handler(**block.input) if handler else "❌ 未知工具"
                    print(f"🔧 {block.name}: {output[:80]}...\n" if len(output) > 80 else f"🔧 {block.name}: {output}\n")
                
                results.append({
                    "type": "tool_result",
                    "tool_use_id": block.id,
                    "content": output,
                })
        
        todo_manager.increment_rounds()
        messages.append({"role": "user", "content": results})

# 测试
if __name__ == "__main__":
    agent_loop("创建一个Python包，包含：1) __init__.py 2) utils.py 3) tests/test_utils.py")
```

## 🎯 模板4：Subagent系统（s04）

添加了子Agent支持。

```python
def run_subagent(prompt: str) -> str:
    """创建一个子Agent来完成任务"""
    sub_messages = [{"role": "user", "content": prompt}]
    
    print(f"  🔄 子Agent启动: {prompt[:50]}...")
    
    for round_num in range(30):
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            system=SYSTEM,
            messages=sub_messages,
            tools=CHILD_TOOLS,  # 子Agent没有 task 工具
            max_tokens=8000,
        )
        
        sub_messages.append({"role": "assistant", "content": response.content})
        
        if response.stop_reason != "tool_use":
            summary = ""
            for block in response.content:
                if hasattr(block, "text"):
                    summary += block.text
            print(f"  ✅ 子Agent完成: {summary[:50]}...")
            return summary or "(无摘要)"
        
        results = []
        for block in response.content:
            if block.type == "tool_use":
                handler = TOOL_HANDLERS.get(block.name)
                output = handler(**block.input) if handler else "❌ 未知工具"
                results.append({
                    "type": "tool_result",
                    "tool_use_id": block.id,
                    "content": str(output)[:50000]
                })
        
        sub_messages.append({"role": "user", "content": results})
    
    return "❌ 子Agent超过最大轮数"

# 定义子Agent的工具（不包含 task）
CHILD_TOOLS = [
    tool for tool in TOOLS if tool["name"] != "task"
]

# 定义主Agent的工具（包含 task）
TASK_TOOL = {
    "name": "task",
    "description": "创建一个子Agent来完成复杂子任务。子Agent有干净的上下文。",
    "input_schema": {
        "type": "object",
        "properties": {
            "prompt": {
                "type": "string",
                "description": "给子Agent的任务描述"
            }
        },
        "required": ["prompt"]
    }
}

PARENT_TOOLS = TOOLS + [TASK_TOOL]

# 在 TOOL_HANDLERS 中添加
TOOL_HANDLERS["task"] = lambda **kw: run_subagent(kw["prompt"])

# 在主Agent的 agent_loop 中修改
def agent_loop(query: str):
    messages = [{"role": "user", "content": query}]
    print(f"\n🤖 主Agent: {query}\n")
    
    while True:
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            system=SYSTEM,
            messages=messages,
            tools=PARENT_TOOLS,  # 使用 PARENT_TOOLS
            max_tokens=8000,
        )
        
        messages.append({"role": "assistant", "content": response.content})
        
        if response.stop_reason != "tool_use":
            for block in response.content:
                if hasattr(block, "text"):
                    print(f"✅ 主Agent: {block.text}\n")
            return
        
        results = []
        for block in response.content:
            if block.type == "tool_use":
                handler = TOOL_HANDLERS.get(block.name)
                output = handler(**block.input) if handler else "❌ 未知工具"
                
                results.append({
                    "type": "tool_result",
                    "tool_use_id": block.id,
                    "content": output,
                })
        
        messages.append({"role": "user", "content": results})

# 测试
if __name__ == "__main__":
    agent_loop("使用子任务来分析项目中的所有Python文件")
```

## 🔧 添加新工具的完整步骤

### 步骤1：定义Schema

```python
MY_TOOL = {
    "name": "my_tool",
    "description": "我的工具的描述",
    "input_schema": {
        "type": "object",
        "properties": {
            "param1": {"type": "string", "description": "参数1"},
            "param2": {"type": "integer", "description": "参数2"}
        },
        "required": ["param1"]
    }
}
```

### 步骤2：实现处理函数

```python
def run_my_tool(param1: str, param2: int = 0) -> str:
    """处理函数"""
    try:
        result = f"处理了 {param1}，参数2={param2}"
        return result
    except Exception as e:
        return f"❌ 错误: {e}"
```

### 步骤3：添加到TOOLS

```python
TOOLS.append(MY_TOOL)
```

### 步骤4：添加到TOOL_HANDLERS

```python
TOOL_HANDLERS["my_tool"] = lambda **kw: run_my_tool(
    kw["param1"],
    kw.get("param2", 0)
)
```

### 步骤5：测试

```python
agent_loop("使用 my_tool 来处理某个任务")
```

## 💡 常见错误

### 错误1：工具没有添加到TOOLS

```python
# ❌ 错误
TOOL_HANDLERS["my_tool"] = lambda **kw: run_my_tool(**kw)
# 忘记添加到 TOOLS

# ✅ 正确
TOOLS.append(MY_TOOL)
TOOL_HANDLERS["my_tool"] = lambda **kw: run_my_tool(**kw)
```

### 错误2：参数名不匹配

```python
# ❌ 错误
"properties": {"param": {"type": "string"}}
# 但处理函数是
def run_tool(parameter: str) -> str:

# ✅ 正确
"properties": {"parameter": {"type": "string"}}
def run_tool(parameter: str) -> str:
```

### 错误3：子Agent有task工具

```python
# ❌ 错误
CHILD_TOOLS = PARENT_TOOLS  # 包含 task

# ✅ 正确
CHILD_TOOLS = [tool for tool in TOOLS if tool["name"] != "task"]
```
