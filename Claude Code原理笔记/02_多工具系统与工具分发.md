> 💡 **核心洞察**：添加工具不需要改循环。只需要一个字典 + 一个查表。

## 🎯 问题：为什么不能只用Bash？

Bash虽然万能，但有问题：

```bash
# ❌ 问题1：不安全
cat /etc/passwd  # 可以读系统文件！

# ❌ 问题2：不可靠
cat huge_file.txt  # 输出可能被截断

# ❌ 问题3：不优雅
sed -i 's/old/new/g' file.txt  # 特殊字符容易出错
```

**解决方案**：用专门的工具替代bash的某些功能。

```
bash（万能但危险）
  ↓
read_file（安全、可靠）
write_file（安全、可靠）
edit_file（精确、可靠）
bash（只用于真正需要的地方）
```

---

## 🔑 核心设计：工具分发（Dispatch）

不改循环，只改工具。怎么做？**用一个字典**。

```python
TOOL_HANDLERS = {
    "bash": run_bash,
    "read_file": run_read_file,
    "write_file": run_write_file,
    "edit_file": run_edit_file,
}

# 使用时：
handler = TOOL_HANDLERS[tool_name]
output = handler(**tool_input)
```

**这就是工具分发的全部秘密。** 一个字典，一个查表。

## 💻 完整代码：s02_tool_use.py

```python
from anthropic import Anthropic
from pathlib import Path
import subprocess

client = Anthropic()

# 工作目录（沙箱）
WORKDIR = Path("./workspace")
WORKDIR.mkdir(exist_ok=True)

SYSTEM = """你是一个编程助手。
用户会给你任务，你需要用工具完成它们。
优先使用 read_file、write_file、edit_file，而不是bash。
只在必要时使用bash。"""

# 定义所有工具的schema
TOOLS = [
    {
        "name": "bash",
        "description": "在终端运行bash命令",
        "input_schema": {
            "type": "object",
            "properties": {
                "command": {"type": "string", "description": "bash命令"}
            },
            "required": ["command"]
        }
    },
    {
        "name": "read_file",
        "description": "读取文件内容",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {"type": "string", "description": "文件路径"}
            },
            "required": ["path"]
        }
    },
    {
        "name": "write_file",
        "description": "创建或覆盖文件",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {"type": "string", "description": "文件路径"},
                "content": {"type": "string", "description": "文件内容"}
            },
            "required": ["path", "content"]
        }
    },
    {
        "name": "edit_file",
        "description": "编辑文件（替换文本）",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {"type": "string", "description": "文件路径"},
                "old_text": {"type": "string", "description": "要替换的文本"},
                "new_text": {"type": "string", "description": "新文本"}
            },
            "required": ["path", "old_text", "new_text"]
        }
    }
]

# ============ 工具实现 ============

def safe_path(p: str) -> Path:
    """确保路径不会逃出工作目录（沙箱）"""
    path = (WORKDIR / p).resolve()
    if not path.is_relative_to(WORKDIR):
        raise ValueError(f"❌ 路径逃出工作目录: {p}")
    return path

def run_bash(command: str) -> str:
    """执行bash命令"""
    try:
        result = subprocess.run(
            command,
            shell=True,
            capture_output=True,
            text=True,
            timeout=10,
            cwd=WORKDIR
        )
        return result.stdout + result.stderr
    except subprocess.TimeoutExpired:
        return "❌ 命令超时"
    except Exception as e:
        return f"❌ 错误: {e}"

def run_read_file(path: str) -> str:
    """读取文件"""
    try:
        content = safe_path(path).read_text(encoding='utf-8')
        return content
    except FileNotFoundError:
        return f"❌ 文件不存在: {path}"
    except Exception as e:
        return f"❌ 错误: {e}"

def run_write_file(path: str, content: str) -> str:
    """创建或覆盖文件"""
    try:
        file_path = safe_path(path)
        file_path.parent.mkdir(parents=True, exist_ok=True)
        file_path.write_text(content, encoding='utf-8')
        return f"✅ 文件已创建: {path}"
    except Exception as e:
        return f"❌ 错误: {e}"

def run_edit_file(path: str, old_text: str, new_text: str) -> str:
    """编辑文件（替换文本）"""
    try:
        file_path = safe_path(path)
        content = file_path.read_text(encoding='utf-8')
        
        if old_text not in content:
            return f"❌ 找不到要替换的文本"
        
        new_content = content.replace(old_text, new_text)
        file_path.write_text(new_content, encoding='utf-8')
        return f"✅ 文件已编辑: {path}"
    except Exception as e:
        return f"❌ 错误: {e}"

# ============ 工具分发 ============

TOOL_HANDLERS = {
    "bash": lambda **kw: run_bash(kw["command"]),
    "read_file": lambda **kw: run_read_file(kw["path"]),
    "write_file": lambda **kw: run_write_file(kw["path"], kw["content"]),
    "edit_file": lambda **kw: run_edit_file(kw["path"], kw["old_text"], kw["new_text"]),
}

# ============ Agent循环 ============

def agent_loop(query: str):
    """Agent主循环"""
    messages = [{"role": "user", "content": query}]
    
    print(f"\n🤖 用户: {query}\n")
    
    while True:
        response = client.messages.create(
            model="claude-3-5-sonnet-20241022",
            system=SYSTEM,
            messages=messages,
            tools=TOOLS,
            max_tokens=8000,
        )
        
        messages.append({"role": "assistant", "content": response.content})
        
        if response.stop_reason != "tool_use":
            for block in response.content:
                if hasattr(block, "text"):
                    print(f"✅ Claude: {block.text}\n")
            return
        
        results = []
        for block in response.content:
            if block.type == "tool_use":
                tool_name = block.name
                tool_input = block.input
                
                print(f"🔧 使用工具: {tool_name}")
                print(f"   参数: {tool_input}")
                
                # 关键：从字典查表
                handler = TOOL_HANDLERS.get(tool_name)
                if handler:
                    output = handler(**tool_input)
                else:
                    output = f"❌ 未知工具: {tool_name}"
                
                print(f"   结果: {output[:100]}...\n" if len(output) > 100 else f"   结果: {output}\n")
                
                results.append({
                    "type": "tool_result",
                    "tool_use_id": block.id,
                    "content": output,
                })
        
        messages.append({"role": "user", "content": results})

# ============ 主程序 ============

if __name__ == "__main__":
    tasks = [
        "读取 requirements.txt 文件",
        "创建一个文件叫 greet.py，内容是一个 greet(name) 函数",
        "编辑 greet.py，给函数添加一个docstring",
        "读取 greet.py 验证编辑成功",
    ]
    
    for task in tasks:
        agent_loop(task)
        print("=" * 60)
```

## 🎮 工具分发的魔法

### 之前（s01）：硬编码

```python
if block.name == "bash":
    output = run_bash(block.input["command"])
elif block.name == "read_file":
    output = run_read_file(block.input["path"])
elif block.name == "write_file":
    output = run_write_file(block.input["path"], block.input["content"])
# ... 更多elif ...
```

**问题**：每添加一个工具，就要加一个elif。代码变得又长又丑。

### 现在（s02）：字典查表

```python
TOOL_HANDLERS = {
    "bash": lambda **kw: run_bash(kw["command"]),
    "read_file": lambda **kw: run_read_file(kw["path"]),
    "write_file": lambda **kw: run_write_file(kw["path"], kw["content"]),
    "edit_file": lambda **kw: run_edit_file(kw["path"], kw["old_text"], kw["new_text"]),
}

handler = TOOL_HANDLERS.get(block.name)
output = handler(**block.input)
```

**优点**：
- ✅ 添加工具只需要加一行
- ✅ 循环代码完全不变
- ✅ 清晰、优雅、可扩展

## 🔒 沙箱：路径安全

最关键的是 `safe_path()` 函数：

```python
def safe_path(p: str) -> Path:
    """确保路径不会逃出工作目录"""
    path = (WORKDIR / p).resolve()
    if not path.is_relative_to(WORKDIR):
        raise ValueError(f"路径逃出工作目录: {p}")
    return path
```

**为什么需要？**

```python
# ❌ 没有沙箱
read_file("../../../etc/passwd")  # 可以读系统文件！

# ✅ 有沙箱
read_file("../../../etc/passwd")  # 错误：路径逃出工作目录
```

**工作原理**：
1. `WORKDIR / p` → 拼接路径
2. `.resolve()` → 解析所有 `..` 和符号链接
3. `.is_relative_to(WORKDIR)` → 检查是否在工作目录内

## 📊 工具对比

| 工具 | 优点 | 缺点 | 何时用 |
|------|------|------|--------|
| **bash** | 万能 | 不安全、不可靠 | 真正需要时 |
| **read_file** | 安全、可靠 | 只能读 | 读文件 |
| **write_file** | 安全、可靠 | 只能写 | 创建文件 |
| **edit_file** | 精确、可靠 | 只能替换 | 编辑文件 |

## 📝 总结

| 概念 | 说明 |
|------|------|
| **TOOL_HANDLERS** | 工具名 → 处理函数的映射 |
| **safe_path()** | 沙箱：防止路径逃出 |
| **lambda** | 匿名函数，用来包装参数 |
| **dispatch** | 根据名字查表调用函数 |
